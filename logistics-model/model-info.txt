***************
CSV DATA FORMAT
***************

JUNCTIONS
file: junctions.csv
format: x coord, y coord
example: 3,5

ROADS
file: roads.csv
format: start x coord, start y coord, end x coord, end y coord
example: 3,5,9,4

STATIONS
file: stations.csv
format: x coordinate, y coordinate
example: 5,8

TRACKS
file: tracks.csv
format: start x coord, start y coord, end x coord, end y coord
example: 5,8,3,2
